-- Создаем новую базу данных для нашего примера
CREATE DATABASE CompanyDB;

-- Переключаем контекст на созданную базу
USE CompanyDB;

-- Создаем таблицу "Сотрудники"
CREATE TABLE Employees (
    EmployeeID INT IDENTITY(1,1) PRIMARY KEY,     -- Уникальный ID сотрудника
    FirstName NVARCHAR(50) NOT NULL,              -- Имя
    LastName NVARCHAR(50) NOT NULL,               -- Фамилия
    Email NVARCHAR(100),                          -- Email
    PhoneNumber NVARCHAR(20),                     -- Номер телефона
    Salary DECIMAL(10, 2) NOT NULL,               -- Зарплата
    HireDate DATE NOT NULL,                       -- Дата приема на работу
    BirthDate DATE NOT NULL                       -- Дата рождения
);

-- Создаем таблицу "Проекты"
CREATE TABLE Projects (
    ProjectID INT IDENTITY(1,1) PRIMARY KEY,      -- Уникальный ID проекта
    ProjectName NVARCHAR(100) NOT NULL,           -- Название проекта
    StartDate DATE NOT NULL,                      -- Дата начала проекта
    EndDate DATE,                                 -- Дата окончания проекта
    Budget DECIMAL(15, 2) NOT NULL                -- Бюджет проекта
);

-- 1. CONCAT - объединение строк
SELECT 
    FirstName,
    LastName,
    CONCAT(FirstName, ' ', LastName) AS FullName,
    CONCAT(LastName, ', ', FirstName) AS LastFirst
FROM Employees;

-- 2. LEN (LENGTH) - длина строки
SELECT 
    FirstName,
    LEN(FirstName) AS NameLength,
    Email,
    LEN(Email) AS EmailLength
FROM Employees;

-- 3. SUBSTRING - извлечение части строки
SELECT 
    Email,
    SUBSTRING(Email, 1, CHARINDEX('@', Email) - 1) AS Username,  -- Имя пользователя до @
    SUBSTRING(Email, CHARINDEX('@', Email) + 1, LEN(Email)) AS Domain  -- Домен после @
FROM Employees;

-- 4. REPLACE - замена части строки
SELECT 
    PhoneNumber,
    REPLACE(PhoneNumber, '+7-', '8-') AS RussianFormat,  -- Замена международного формата
    REPLACE(Email, 'company.com', 'ourfirm.ru') AS NewEmail  -- Замена домена
FROM Employees;

-- 5. Комплексный пример строковых функций
SELECT 
    FirstName,
    LastName,
    Email,
    CONCAT(UPPER(SUBSTRING(FirstName, 1, 1)), LOWER(SUBSTRING(LastName, 1, 3)), 
           EmployeeID) AS SystemLogin,  -- Генерация логина: первая буква имени + 3 буквы фамилии + ID
    REPLACE(PhoneNumber, '-', '') AS CleanPhone  -- Удаление дефисов из номера
FROM Employees;
